<script>
const SUPABASE_URL="https://xphczwydrcuvqsuthiii.supabase.co";
const SUPABASE_KEY="YOUR_ANON_KEY_HERE";
const db=supabase.createClient(SUPABASE_URL,SUPABASE_KEY);

async function load(){

let doc=document.getElementById("doc").value;

const {data}=await db.from("appointments").select("*");

let filtered=data.filter(a=>a.doctor===doc);

let list=document.getElementById("list");
list.innerHTML="";

filtered.forEach(a=>{
let div=document.createElement("div");
div.className="card";

div.innerHTML=`
<p>${a.name} - ${a.date} ${a.time}</p>
<textarea id="n${a.id}">${a.notes||""}</textarea>
<button onclick="save(${a.id})">Save</button>
`;

list.appendChild(div);
});
}

async function save(id){
let note=document.getElementById("n"+id).value;
await db.from("appointments").update({notes:note}).eq("id",id);
alert("Saved");
}
